# 1xxx Prepare Religious Mission
###
missionary.1000 = {
    type = character_event
    theme = faith
    desc = "The Missionaries are requesting some initial funds to begin with the religious mission."
    option = {
        name = "Let us show them that we lack for nothing"
        scope:owner = {
            remove_short_term_gold = massive_gold_value
            add_piety = 50
        }
        scope:scheme = {
            add_scheme_modifier = {
                type = missionary_rich_arrival
            }
        }
    }
    option = {
        name = "They shall have it!"
        scope:owner = {
            remove_short_term_gold = medium_gold_value
            add_piety = 15
        }
    }
    option = {
        name = "It is their Duty to do it!"
        scope:owner = {
            add_piety = -250
            add_prestige = -100
        }
        scope:scheme = {
            add_scheme_modifier = {
                type = missionary_poor_arrival
            }
        }
    }
    option = {
        name = "This is too expensive!"
        scope:scheme = {
            end_scheme = yes
        }
    }
}

# 2000 - Target found interest in Religion
missionary.2000 = {
    type = character_event
    theme = faith
    desc = "The Target is really interested in the true Faith"
    trigger = {
        scope:scheme = {
            NOT = { has_variable = had_missionary_2000_event }
        }
    }
    immediate = {
        scope:scheme = {
            set_variable = {
                name = had_missionary_2000_event
                value = yes
            }
        }
    }
    option = {
        name = "Sounds promising!"
        scope:owner = {
            add_piety = 10
        }
        scope:scheme_priest = {
            add_piety = 35
        }
        scope:scheme = {
            add_scheme_modifier = {
                type = missionary_target_interested_modifier
            }
        }
    }
    option = {
        name = "Then let us push it even further!"
        scope:owner = {
            add_piety = 15
            remove_short_term_gold = activity_medium_gold_value
        }
        scope:scheme_priest = {
            add_piety = 50
        }
        scope:scheme = {
            add_scheme_progress = 2
            add_scheme_modifier = {
                type = missionary_target_interested_modifier
            }
        }
    }
}

# 4000 - You are thrown out
missionary.4000 = {
    type = character_event
    theme = faith
    desc = "Your Religious Mission was thrown out of the Country. And if you come back there may be consequences."
    immediate = {
        scope:scheme = {
            add_scheme_progress = -3
        }
    }
    option = {
        name = "Time to go home"
        scope:scheme = {
            end_scheme = yes
        }
    }
    option = {
        name = "There is no going back!"
        random_list = {
            10 = {
                scope:scheme_priest = {
                        death = {
                            killer = scope:target
                            death_reason = death_execution
                        }
                }
                scope:scheme = {
                    end_scheme = yes
                }
            }
            50 = {
                rightfully_imprison_character_effect = {
                	TARGET = scope:scheme_priest
                	IMPRISONER = scope:target
                }
                scope:scheme = {
                    end_scheme = yes
                }
            }
            200 = {
                scope:scheme = {
                    end_scheme = yes
                }
            }
            100 = {
                scope:owner = {
                    add_piety = 15
                }
                scope:scheme_priest = {
                    add_piety = 50
                }
            }
        }
    }
}
# 4001 - Your Priest insulted the target
missionary.4001 = {
    type = character_event
    theme = faith
    desc = "Your Priest has lost his nerves and insulted all non (yet) believers"
    immediate = {
        add_scheme_progress = -2
        scope:scheme_priest = {
            add_prestige = -50
            add_piety = -15
        }
        scope:scheme = {
            set_variable = {
                name = had_missionary_4001_event
                value = yes
            }
            add_scheme_modifier = {
                type = missionary_target_insulted_modifier
            }
        }
    }
    trigger = {
        scope:scheme = {
            NOT = { has_variable = had_missionary_4001_event }
        }
    }
    option = {
        name = "You idiot!"
    }
}

# 9000 The landed target converts with his vassals and countries
missionary.9000 = {
    type = character_event
    theme = faith
    desc = "A whole Realm found into the Light."

    left_portrait = scope:target

    option = {
        name = "Welcome to the Light!"

        scope:target = {
            save_scope_as = recipient
        }

        scope:owner = {
            save_scope_as = actor
        }

        scope:target = {
            missionary_convert_target = yes
            missionary_convert_spouses = yes
            missionary_convert_family_in_court = yes
        }

        scope:target = {
            missionary_convert_courtiers = yes
            missionary_convert_vassal = yes
            missionary_convert_province = yes
        }

        scope:owner = {
            add_piety = 250
        }

        scope:scheme = {
            end_scheme = yes
        }
    }
}
